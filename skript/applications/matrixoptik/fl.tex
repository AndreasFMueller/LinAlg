        \fill[color=white] #1 circle[radius=0.08];

	({-\d},{\h}) arc (180-\a:180+\a:\R)
	--
	({\d},{-\h}) arc (-\a:+\a:\R)
	--cycle;

	({-\d},{\h}) arc (180-\a:180+\a:\R)
	--
	({\d},{-\h}) arc (-\a:+\a:\R)
	--cycle;
