	\draw[connection](begin)--(s0);
	\draw[connection](s0)--node[right,yshift=2pt]{\footnotesize Ja}(s1);

	\draw[connection](s1)--node[above]{\footnotesize Nein}(s3);
	\draw[connection](s1)--node[right,yshift=2pt]{\footnotesize Ja}(s4);
