\def\punkt#1{
        \fill[color=white] #1 circle[radius=0.08];
        \draw #1 circle[radius=0.08];
}

\begin{tikzpicture}[>=latex,thick]
\coordinate (O) at (0,0);
\coordinate (P) at (6,1);
\draw[->] (O)--(P);
\punkt{(O)} \node at (O) [left] {$O$};
\punkt{(P)} \node at (P) [right] {$P$};
\end{tikzpicture}
