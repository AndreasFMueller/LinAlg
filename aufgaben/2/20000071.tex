Berechnen Sie die Determinante des magischen Quadrates
\[
M
=
\begin{pmatrix}
   2 &  7 &  6 \\
   9 &  5 &  1 \\
   4 &  3 &  8
\end{pmatrix}.
\]

\begin{loesung}
Nach der Sarrus-Formel ist
\begin{align*}
\det M
&=
  2\cdot 5\cdot 8 + 7\cdot 1 \cdot 4 + 6\cdot 9\cdot 3
- 4\cdot 5\cdot 6 - 3\cdot 1 \cdot 2 - 8\cdot 9\cdot 7
\\
&=
80 + 28 + 162 - 120 - 6 - 504
\\
&=270 - 630
=
-360.
\qedhere
\end{align*}
\end{loesung}
