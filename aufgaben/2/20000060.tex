\[
A
=
\begin{pmatrix*}[r]

Wir entwickeln nach der ersten  Spalte und erhalten
\begin{align*}

10\cdot

  -1 &   -9 &    7  \\

-1\cdot
\left|\,\begin{matrix*}[r]
   9 &    1 &    3  \\
   2 &   -8 &   -3  \\
  -1 &   -9 &    7 
\end{matrix*}\,\right|.

\end{loesung}
