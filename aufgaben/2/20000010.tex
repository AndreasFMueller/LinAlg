Berechnen Sie die Determinante
\[
\delta=\left|\,
\begin{matrix}
a_{11}&a_{12}&0     &0     \\
a_{21}&a_{22}&0     &0     \\
0     &0     &d_{11}&d_{12}\\
0     &0     &d_{21}&d_{22}
\end{matrix}
\,\right|
\]

\thema{Determinante}
\thema{Entwicklungssatz}

\begin{loesung}
Entwicklung nach der ersten Zeile ergibt
\begin{align*}
\delta
&=
a_{11}\left|\,
\begin{matrix}
a_{22}&0     &0     \\
0     &d_{11}&d_{12}\\
0     &d_{21}&d_{22}
\end{matrix}
\,\right|
-
a_{21}
\left|\,
\begin{matrix}
a_{12}&0     &0     \\
0     &d_{11}&d_{12}\\
0     &d_{21}&d_{22}
\end{matrix}
\,\right|
\\
&=
a_{11}a_{22}\left|\,
\begin{matrix}
d_{11}&d_{12}\\
d_{21}&d_{22}
\end{matrix}
\,\right|
-
a_{21}a_{12}
\left|\,
\begin{matrix}
d_{11}&d_{12}\\
d_{21}&d_{22}
\end{matrix}
\,\right|
\\
&=
\left|\,
\begin{matrix}
a_{11}&a_{12}\\
a_{21}&a_{22}
\end{matrix}
\,\right|
\cdot
\left|\,
\begin{matrix}
d_{11}&d_{12}\\
d_{21}&d_{22}
\end{matrix}
\,\right|
\\
&=\det(A)\det(D),
\end{align*}
wobei wir mit $A$ und $D$ die beiden $2\times 2$ in den diagonalen Ecken
der Determinante meinen, also
\[
A=\begin{pmatrix}
a_{11}&a_{12}\\a_{21}&a_{22}
\end{pmatrix},\qquad
D=\begin{pmatrix}
d_{11}&d_{12}\\d_{21}&d_{22}
\end{pmatrix}.
\qedhere
\]
\end{loesung}
