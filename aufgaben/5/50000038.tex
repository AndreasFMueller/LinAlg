Berechnen Sie das Vektorprodukt $\vec{a}\times\vec{b}$ der Vektoren
\[
\vec{a}
=
\begin{pmatrix}
1\\1\\0
\end{pmatrix}
\qquad\text{und}\qquad
\vec{b}
=
\begin{pmatrix}
0\\1\\1
\end{pmatrix}
\]

\begin{loesung}
Das Vektorprodukt ist
\begin{align*}
\vec{a}\times\vec{b}
&=
\begin{pmatrix}
1\\1\\0
\end{pmatrix}
\times
\begin{pmatrix}
0\\1\\1
\end{pmatrix}
=
\begin{pmatrix}
1\cdot 1 - 0\cdot 1 \\
0\cdot 1 - 1\cdot 1 \\
1\cdot 1 - 0\cdot 1
\end{pmatrix}
=
\begin{pmatrix*}[r]
 1\\
-1\\
 1
\end{pmatrix*}.
\qedhere
\end{align*}
\end{loesung}
