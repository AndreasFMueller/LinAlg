Hat das folgende Gleichungssystem eine L"osung?
$$
\begin{linsys}{2}
 3x & + &  2y & = &  5 \\
  x & + &  3y & = &  4 \\
 7x & + & 11y & = & 19
\end{linsys}
$$

\ifthenelse{\boolean{loesungen}}{
\begin{loesung}
Wir verwenden das Gauss-Verfahren.
\begin{align*}
\begin{tabular}{|cc|c|}
\hline
3&2&5\\
1&3&4\\
7&11&19\\
\hline
\end{tabular}
&\rightarrow
\begin{tabular}{|cc|c|}
\hline
1&3&4\\
3&2&5\\
7&11&19\\
\hline
\end{tabular}
\\
&\rightarrow
\begin{tabular}{|>{$}c<{$}>{$}c<{$}|>{$}c<{$}|}
\hline
1&3&4\\
0&-7&-7\\
0&-10&-9\\
\hline
\end{tabular}
\\
&\rightarrow
\begin{tabular}{|>{$}c<{$}>{$}c<{$}|>{$}c<{$}|}
\hline
1&3&4\\
0&1&1\\
0&-10&-9\\
\hline
\end{tabular}
\\
&\rightarrow
\begin{tabular}{|cc|c|}
\hline
1&3&4\\
0&1&1\\
0&0&1\\
\hline
\end{tabular}
\\
\end{align*}
Die letzt Zeile entspricht der Gleichung $0=1$, also ein Widerspruch.
Dieses Gleichungssystem kann keine L"osung haben.
\end{loesung}
}{}
