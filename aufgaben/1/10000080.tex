Schreiben Sie das Gleichungssystem
\begin{equation*}
\renewcommand{\arraycolsep}{3pt}
\begin{array}{rcrcr}
4x &+& 7y &=& 9 \\
3x &+& 5y &=& 4
\end{array}
\end{equation*}
in Tableau-Form.

\begin{loesung}
Die Tableau-Form ist
\begin{equation*}
\begin{tabular}{|>{$}r<{$}>{$}r<{$}|>{$}r<{$}|}
\hline
x_1&x_2&1\\
\hline
4&7&9 \\
3&5&4 \\
\hline
\end{tabular}
\qedhere
\end{equation*}
\end{loesung}
