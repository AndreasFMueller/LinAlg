Schreiben Sie das Gleichungssystem
\begin{equation*}
\renewcommand{\arraycolsep}{3pt}
\begin{array}{rcrcr}
4x_1 &+& 7x_2 &=& 9 \\
3x_1 &+& 5x_2 &=& 4
\end{array}
\end{equation*}
in Tableau-Form.

\begin{loesung}
Die Tableau-Form ist
\begin{equation*}
\raisebox{0.6cm}{%
\begin{tabular}{|>{$}r<{$}>{$}r<{$}|>{$}r<{$}|}
\hline
x_1&x_2&1\\
\hline
4&7&9 \\
3&5&4 \\
\hline
\end{tabular}}
\qedhere
\end{equation*}
\end{loesung}
