L"osen Sie die folgenden Gleichungssysteme und geben Sie an,

\begin{teilaufgaben}
\item
\[
\begin{linsys}{2}
 x & + & 3y & = &  1\\
-x & + & 5y & = & -1
\end{linsys}
\]
\item
\[
\begin{linsys}{3}
x & + & 2y & + &  z & = & 1\\
  &   &  y & + & 2z & = & 3\\
x & + & 2y & + & 2z & = & 5 
\end{linsys}
\]
\item
\[
\begin{linsys}{2}
  x & - & 3y & = & 4 \\
-2x & + & 6y & = & 5
\end{linsys}
\]
\item
\[
\begin{linsys}{2}
  x & - & 3y & =  4 \\
-2x & + & 6y & = -8
\end{linsys}
