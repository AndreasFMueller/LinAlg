In der Ebene sind die Punkte $A$ bis $D$ gegeben.
Finden Sie ihre Koordinaten in der Basis $\vec{b}_1$ und $\vec{b}_2$.
\begin{center}
\definecolor{darkred}{rgb}{0.8,0,0}
\begin{tikzpicture}[>=latex,thick,scale=0.5]
\coordinate (b1) at (3,-2);
\coordinate (b2) at (1,2);

\ifthenelse{\boolean{loesungen}}{
\begin{scope}
\clip (-5.1,-4.1) rectangle (8.1,6.1);
\foreach \x in {-2,...,3}{
	\draw[color=darkred,line width=0.3pt]
		($\x*(b1)-3*(b2)$) -- ++($8*(b2)$);
}
\foreach \y in {-3,...,5}{
	\draw[color=darkred,line width=0.3pt]
		($-2*(b1)+\y*(b2)$) -- ++($5*(b1)$);
}
\end{scope}
}{}

\foreach \x in {-5,...,8}{
	\draw[color=gray!20] (\x,-4.1) -- (\x,6.3);
}
\foreach \y in {-4,...,6}{
	\draw[color=gray!20] (-5.1,\y) -- (8.3,\y);
}

\draw[->,color=darkred,line width=1.4pt] (0,0) -- (b1);
\node[color=darkred] at ($1.0*(b1)$) [below] {$\vec{b}_1$};

\draw[->,color=darkred,line width=1.4pt] (0,0) -- (b2);
\node[color=darkred] at ($1.0*(b2)$) [right] {$\vec{b}_2$};

\draw[->] (-5.1,0) -- (8.5,0);
\draw[->] (0,-4.1) -- (0,6.5);

\fill[color=blue] ($1*(b1)+3*(b2)$) circle[radius=0.16];
\node[color=blue] at ($1*(b1)+3*(b2)+(0.1,-0.2)$) [above right] {$A$};

\fill[color=blue] ($-1*(b1)+2*(b2)$) circle[radius=0.16];
\node[color=blue] at ($-1*(b1)+2*(b2)+(-0.1,0.4)$) [below left] {$B$};

\fill[color=blue] ($2*(b1)+1*(b2)$) circle[radius=0.16];
\node[color=blue] at ($2*(b1)+1*(b2)+(0.1,-0.4)$) [above right] {$C$};

\fill[color=blue] ($-1*(b1)-1*(b2)$) circle[radius=0.16];
\node[color=blue] at ($-1*(b1)-1*(b2)+(0,0.2)$) [below left] {$D$};

\end{tikzpicture}
\end{center}
